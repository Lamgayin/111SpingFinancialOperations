#實驗目標：實驗上述過程模擬10,0000次，驗證是否真有4850次是100次內贏55次
#說明    ：getExperimentResult() 返回模擬10,0000次實驗內贏55次的次數
#假設    ：出現人頭(值為1，贏) / 出現數字(值為0，輸)

getExperimentResult <- function(){
  count <-0                                 #初始一個累積數量的變數，「即實驗10,0000次內贏55次的累計數量」
  
  for(i in 1:100000){                       #將實驗重複10,0000次
    result <- sample(0:1,100,replace=TRUE)  #抽取100次的0和1的結果
    if(sum(result==1)==55) count <- count+1 #當出現「100次內贏55次」的情形，累計增加1次
  }
  
  return (count)                            #返回實驗結果
}

getExperimentResult()
## [1] 4818
#將上述實驗進行100次，觀察統計結果
count <- 100                                     #實驗次數初始化
result <- c()                                    #實驗結果

for(i in 1:count){                               #將實驗重複count次
  result <- append(result,getExperimentResult()) #將實驗結果添加進result
}

#統計結果
range(result)     #10,0000次內100次內贏55次的次數範圍
## [1] 4648 5062
sum(result==4850) #剛好等於4850次的次數數量
## [1] 1
summary(result)   #其他統計訊息
##    Min. 1st Qu.  Median    Mean 3rd Qu.    Max. 
##    4648    4799    4848    4848    4899    5062
plot(x=1:100,y=result,col="blue",type="o") #繪製「10,0000次內100次內贏55次的次數」圖表

"結論：根據統計訊息以及圖表顯示"

"實驗10,0000次，100次內贏55次不一定會剛好是4850次，但它的次數範圍會趨近於這個數

"存在一定程度的誤差，但是趨勢相近的"
